```C
int sqrt(int);

{

}

int sqrt(int num) {
    int i = 1;
    int res;

    for (i = 1; i < num; i++)
    {
        res = i * i;
            if (res == num)
            {
                res = i;
                break;
            } else
            {
                res = 0;
            }
    }

    return res;
}
```
