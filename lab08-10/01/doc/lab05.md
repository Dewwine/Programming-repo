2.4 Написання коду і коментарів до нього

```C
int nod(int, int);

int get_rand();

int main()
{
  srand(time(0));

  nod(6, get_rand());
  return 0;
}

int nod(int a, int b)
{
  int res;
  int i = 1;

  if (a < b)
  {
    int x = a;
    a = b;
    b = x;
  }

  for (i = 1; i <= b; i++)
  {
    if ((a % i) == 0 && (b % i) == 0)
    {
      res = i;
    }
  }

  return res;
}

int get_rand() {
  int random = rand() % 10 + 1; 
  return random;
}
```
