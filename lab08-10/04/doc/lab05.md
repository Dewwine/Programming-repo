2.1 Створення файлу з вихідним кодом, написання коду і коментарів до нього

```C
int isPerfect(int);

int main()
{
    isPerfect(28);
	return 0;
}

int isPerfect(int n) 
{
    int count = 0;
    int res = 0;
    int i = 1;

    for (i = 1; i <= n/2; i++)
    {
        if (n % i == 0)
        {
            count += i;
        }
    }

    if (n == count)
    {
        res = 1;
    } else
    {
        res = 0;
    }

    return res;
}
```
