2.1 Створення файлу з вихідним кодом, написання коду і коментарів до нього

```C
const int N = 28;

void fillSimple(int);

int main()
{
  fillSimple(N);
  return 0;
}

void fillSimple(int size)
{
  int arr[size];
  int stop = 0;
  int i = 0;
  int num = 2;

  while (stop < 1)
  {
    for (int j = 2; j <= num; j++)
    {
      if (num % j == 0)
      {
        if (num == j)
        {
          arr[i] = num;
          i++;
        }
        else
        {
          break;
        }
      }
    }
    num++;
    if (i == N)
    {
      stop++;
    }
  }
}
```
