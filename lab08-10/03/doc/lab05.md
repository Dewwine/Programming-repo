```C
int isLucky(int);

int main()
{    
    printf("%d\n", isLucky(102300));
	return 0;
}

int isLucky(int n) {
    int x1 = 0, x2 = 0, i = 0;
    int res;

    for(i=0; i<3; i++)
    {
        x2 += n % 10; 
        n /= 10;
    }

    for(i=0; i<3; i++)
    {
        x1 += n % 10;
        n /= 10;
    }

    if (x1 == x2)
    {
        res = 1;
    } else
    {
        res = 0;
    }
    return res;
}
```
